新增使用者
----------

    [root@Parabola ~]# pacman -S sudo
    [root@Parabola ~]# useradd -m -G wheel -s /bin/bash bill    # bill 改成你的使用者 
    [root@Parabola ~]# passwd bill    # bill 改成你的使用者
    [root@Parabola ~]# visudo
    root ALL=(ALL) ALL
    bill ALL=(ALL) ALL
